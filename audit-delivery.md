<script>
  (function() {
    const STORAGE_KEY = "citadelAudit";
    const fields = document.querySelectorAll("[data-storage]");
    if (!fields.length) return;

    const params = new URLSearchParams(window.location.search);
    const paramUpdates = {};
    ["name", "email", "phone"].forEach((key) => {
      const value = params.get(key);
      if (value) {
        paramUpdates[key] = value;
      }
    });

    let stored = {};
    try {
      stored = JSON.parse(localStorage.getItem(STORAGE_KEY) || "{}");
    } catch (_) {
      stored = {};
    }

    if (Object.keys(paramUpdates).length) {
      stored = { ...stored, ...paramUpdates };
      localStorage.setItem(STORAGE_KEY, JSON.stringify(stored));
    }

    if (!stored.email || !stored.phone) {
      window.location.href = "{{ '/free-audit/' | url }}";
      return;
    }

    function mergeStored(updates = {}) {
      stored = { ...stored, ...updates };
      localStorage.setItem(STORAGE_KEY, JSON.stringify(stored));
    }

    fields.forEach(field => {
      const key = field.dataset.storage;
      if (stored[key]) {
        field.value = stored[key];
      }
      field.addEventListener("input", () => mergeStored({ [key]: field.value }));
    });

    const submitBtn = document.getElementById("submit-audit");
    const netlifyForm = document.getElementById("audit-netlify-form");
    const NEXT_URL = "{{ '/business/' | url }}";

    function collectFieldValues() {
      const values = {};
      fields.forEach(field => {
        values[field.dataset.storage] = field.value.trim();
      });
      return values;
    }

    if (submitBtn && netlifyForm) {
      submitBtn.addEventListener("click", () => {
        const auditValues = collectFieldValues();
        mergeStored(auditValues);

        const identity = {
          name: stored.name || "",
          email: stored.email || "",
          phone: stored.phone || ""
        };

        if (!identity.name || !identity.email || !identity.phone) {
          alert("Please complete your name, email, and phone on the audit intake form before submitting.");
          window.location.href = "{{ '/free-audit/' | url }}";
          return;
        }

        // Fill hidden inputs for Netlify
        Object.entries({ ...auditValues, ...identity }).forEach(([key, value]) => {
          const input = netlifyForm.elements.namedItem(key);
          if (input) {
            input.value = value || "";
          }
        });

        const formData = new FormData(netlifyForm);
        const encoded = new URLSearchParams();
        for (const [key, value] of formData.entries()) {
          encoded.append(key, value);
        }

        submitBtn.disabled = true;
        submitBtn.textContent = "Submitting...";

        fetch("{{ '/' | url }}", {
          method: "POST",
          headers: { "Content-Type": "application/x-www-form-urlencoded" },
          body: encoded.toString()
        }).then(() => {
          window.location.href = NEXT_URL;
        }).catch((error) => {
          alert("Submission failed: " + error);
          submitBtn.disabled = false;
          submitBtn.textContent = "Start Your $499 Business Diagnostic";
        });
      });
    }
  })();
</script>
